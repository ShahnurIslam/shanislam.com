  def listSplit[T](collection:Seq[T],seperator:T):Seq[Seq[T]]= {
    val name = ListBuffer(ListBuffer[T]())
    collection foreach {e =>
      if(e == seperator){
        name += ListBuffer[T]()
      } else {
        name.last += e
      }
    }
    name.map(_.toSeq).toSeq
  }
